  36 def CheckFileExt(FileName, FileExts):
  37     """Check file type based on the specified file extensions delimited by spaces.
  38     
  39     Arguments:
  40         FileName (str): Name of a file.
  41         FileExts (str): Space delimited string containing valid file extensions.
  42 
  43     Returns:
  44         bool : True, FileName contains a valid file extension; Otherwise, False.
  45 
  46     """
  47     
  48     for FileExt in FileExts.split():
  49         if re.search(r"\.%s$" % FileExt, FileName, re.IGNORECASE):
  50             return True
  51     
  52     return False

  54 def CheckTextValue(Value, ValidValues):
  55     """Check text value based on the specified valid values delimited by spaces.
  56 
  57     Arguments:
  58         Value (str): Text value
  59         ValidValues (str): Space delimited string containing valid values.
  60 
  61     Returns:
  62         bool : True, Value is valid; Otherwise, False.
  63 
  64     """
  65     
  66     ValidValues = re.sub(' ', '|', ValidValues)
  67     if re.match("^(%s)$" % ValidValues, Value, re.IGNORECASE):
  68         return True
  69     
  70     return False

  72 def GetTextLinesWords(TextFilePath, Delimiter, QuoteChar, IgnoreHeaderLine):
  73     """Parse lines in the specified text file into words in a line and return a list containing
  74     list of parsed line words.
  75 
  76     Arguments:
  77         TextFilePath (str): Text file name including file path.
  78         Delimiter (str): Delimiter for parsing text lines.
  79         QuoteChar (str): Quote character for line words.
  80         IgnoreHeaderLine (bool): A flag indicating whether to ignore first
  81             valid data line corresponding to header line.
  82 
  83     Returns:
  84         list : A list of lists containing parsed words for lines.
  85 
  86     Notes:
  87         The lines starting with # or // are considered comment lines and are
  88         ignored during parsing along with any empty lines.
  89 
  90     """
  91     TextFile = open(TextFilePath, "r")
  92     if TextFile is None:
  93         PrintError("Couldn't open text file: %s.\n" % (TextFilePath))
  94 
  95     # Collect text lines...
  96     TextLines = []
  97     FirstValidLine = True
  98     for Line in TextFile:
  99         Line = Line.strip()
 100         
 101         # Ignore empty lines...
 102         if not len(Line):
 103             continue
 104         
 105         # Ignore comments...
 106         if re.match("^(#|\/\/)", Line, re.I):
 107             continue
 108 
 109         # Ignore header line...
 110         if FirstValidLine:
 111             FirstValidLine = False
 112             if IgnoreHeaderLine:
 113                 continue
 114         
 115         TextLines.append(Line)
 116         
 117     TextFile.close()
 118 
 119     # Parse text lines...
 120     TextLinesWords = []
 121     
 122     TextLinesReader = csv.reader(TextLines, delimiter = Delimiter, quotechar = QuoteChar)
 123     for LineWords in TextLinesReader:
 124         TextLinesWords.append(LineWords)
 125     
 126     return TextLinesWords

 128 def DoesSMILESFileContainTitleLine(FileName):
 129     """Determine whether the SMILES file contain a title line based on the  presence
 130     of a string SMILES, Name or ID in the first line.
 131 
 132     Arguments:
 133         FileName (str): Name of a file.
 134 
 135     Returns:
 136         bool : True, File contains title line; Otherwise, False.
 137 
 138     """
 139     
 140     Infile = open(FileName, "r")
 141     if Infile is None:
 142         return False
 143 
 144     Line = Infile.readline()
 145     Infile.close()
 146 
 147     if re.search("(SMILES|Name|ID)", Line, re.I):
 148         return True
 149         
 150     return False

 152 def GetExamplesTextFromDocOptText(DocOptText):
 153     """Get script usage example lines from a docopt doc string. The example text
 154     line start from a line containing `Examples:`  keyword at the beginning of the line.
 155     
 156     Arguments:
 157         DocOptText (str): Doc string containing script usage examples lines starting with
 158             a line marked by `Examples:` keyword at the beginning of a line.
 159 
 160     Returns:
 161         str : A string containing text lines retrieved from the examples section of
 162             DocOptText parameter.
 163 
 164     """
 165     
 166     ExamplesStart = re.compile("^Examples:", re.IGNORECASE)
 167     ExamplesEnd = re.compile("^(Author:|See also:|Copyright:)", re.IGNORECASE)
 168     
 169     ExamplesText = 'Examples text is not available';
 170     ExamplesTextFound = False
 171     
 172     for Line in DocOptText.splitlines():
 173         if ExamplesStart.match(Line):
 174             ExamplesText = 'Examples:';
 175             ExamplesTextFound = True
 176             continue
 177         
 178         if ExamplesEnd.match(Line):
 179             break
 180         
 181         if ExamplesTextFound:
 182             ExamplesText += "\n" + Line;
 183     
 184     return ExamplesText

 186 def GetExcelStyleColumnLabel(ColNum):
 187     """Return Excel style column label for a colum number.
 188     
 189     Arguments:
 190         ColNum (int): Column number
 191 
 192     Returns:
 193         str : Excel style column label.
 194 
 195     """
 196     Letters = 'ABCDEFGHIJKLMNOPQRSTUVWXYZ'
 197     
 198     ColLabelList = []
 199     while ColNum:
 200         ColNum, SubColNum = divmod(ColNum - 1, 26)
 201         ColLabelList[:0] = Letters[SubColNum]
 202     
 203     return ''.join(ColLabelList)

 505 def ValidateOptionNumberValue(OptionName, OptionValue, CmpOpValueMap):
 506     """Validate option value using comparison operater and value pairs in specified in
 507     a map.
 508     
 509     Arguments:
 510         OptionName (str): Command line option name.
 511         OptionValue (int or float): Command line option value.
 512         CmpOpValueMap (dictionary): Comparison operator key and value pairs to
 513             validate values specified in OptionValue.
 514 
 515     Notes:
 516         The function exits with an error message for an invalid option values specified
 517         in OptionValue.
 518 
 519     Examples:
 520 
 521         ValidateOptionNumberValue("--maxConfs", int(Options["--maxConfs"]),
 522             {">": 0})
 523         ValidateOptionNumberValue("-b, --butinaSimilarityCutoff", 
 524             float(Options["--butinaSimilarityCutoff"]),
 525             {">": 0.0, "<=" : 1.0})
 526 
 527     """
 528     
 529     Status = True
 530     for CmpOp in CmpOpValueMap:
 531         Value = CmpOpValueMap[CmpOp]
 532         if re.match("^>$", CmpOp, re.I):
 533             if OptionValue <= Value:
 534                 Status = False
 535                 break
 536         elif re.match("^>=$", CmpOp, re.I):
 537             if OptionValue < Value:
 538                 Status = False
 539                 break
 540         elif re.match("^<$", CmpOp, re.I):
 541             if OptionValue >= Value:
 542                 Status = False
 543                 break
 544         elif re.match("^<=$", CmpOp, re.I):
 545             if OptionValue > Value:
 546                 Status = False
 547                 break
 548         else:
 549             PrintError("The specified comparison operator, %s, for function MiscUtil.ValidateOptionNumberValue is not supported\n" % (CmpOp))
 550     
 551     if not Status:
 552         FirstValue = True
 553         SupportedValues = ""
 554         for CmpOp in CmpOpValueMap:
 555             Value = CmpOpValueMap[CmpOp]
 556             if FirstValue:
 557                 FirstValue = False
 558                 SupportedValues = "%s %s" % (CmpOp, Value)
 559             else:
 560                 SupportedValues = "%s and %s %s" % (SupportedValues, CmpOp, Value)
 561         
 562         PrintError("The value specified, %s, for option \"%s\" is not valid. Supported value(s): %s " % (OptionValue, OptionName, SupportedValues))
 563 
 564 def ValidateOptionNumberValues(OptionName, OptionValueString, OptionValueCount, OptionValueDelimiter, OptionValueType, CmpOpValueMap):
 565     """Validate numerical option values using option value string, delimiter, value type,
 566     and a specified map containing comparison operator and value pairs.
 567     
 568     Arguments:
 569         OptionName (str): Command line option name.
 570         OptionValueString (str): Command line option value.
 571         OptionValueCount (int): Number of values in OptionValueString.
 572         OptionValueDelimiter (str): Delimiter used for values in OptionValueString.
 573         OptionValueType (str): Valid number types (integer or float)
 574         CmpOpValueMap (dictionary): Comparison operator key and value pairs to
 575             validate values specified in OptionValueString.
 576 
 577     Notes:
 578         The function exits with an error message for invalid option values specified
 579         in OptionValueString
 580 
 581     Examples:
 582 
 583         ValidateOptionNumberValues("-m, --molImageSize",
 584             Options["--molImageSize"], 2, ",", "integer", {">": 0})
 585 
 586     """
 587     if not CheckTextValue(OptionValueType, "integer float"):
 588         PrintError("The option value type specified, %s, for function MiscUtil.ValidateOptionNumberValues  is not valid. Supported value: integer float " % (OptionValueType))
 589         
 590     Values = OptionValueString.split(OptionValueDelimiter)
 591     if len(Values) != OptionValueCount:
 592         PrintError("The value specified, %s, for option \"%s\" is not valid. It must contain %d %s values separated by \"%s\"" % (OptionValueString, OptionName, OptionValueCount, OptionValueType, OptionValueDelimiter))
 593 
 594     IsIntergerValue = True
 595     if re.match("^float$", OptionValueType, re.I):
 596         IsIntergerValue = False
 597     
 598     for Value in Values:
 599         if IsIntergerValue:
 600             if not IsInteger(Value):
 601                 PrintError("The value specified, %s, for option \"%s\" in string \"%s\" must be an integer." % (Value, OptionName, OptionValueString))
 602             Value = int(Value)
 603         else:
 604             if not IsFloat(Value):
 605                 PrintError("The value specified, %s, for option \"%s\" in string \"%s\" must be a float." % (Value, OptionName, OptionValueString))
 606             Value = float(Value)
 607         ValidateOptionNumberValue(OptionName, Value, CmpOpValueMap)

 720     """
 721     if re.match("^auto$", ParamsOptionValue, re.I):
 722         # No specific parameters to process except for parameters with possible auto value...
 723         _ProcessInfileAndOutfileAutoParameters(Mode, ParamsInfo, ParamsOptionName, ParamsOptionValue, InfileName, OutfileName)
 724         return
 725     
 726     ParamsOptionValue = re.sub(" ", "", ParamsOptionValue)
 727     if not ParamsOptionValue:
 728         PrintError("No valid parameter name and value pairs specified using \"%s\" option" % ParamsOptionName)
 729     
 730     ParamsOptionValueWords = ParamsOptionValue.split(",")
 731     if len(ParamsOptionValueWords) % 2:
 732         PrintError("The number of comma delimited paramater names and values, %d, specified using \"%s\" option must be an even number." % (len(ParamsOptionValueWords), ParamsOptionName))
 733         
 734     # Setup a canonical paramater names...
 735     ValidParamNames = []
 736     CanonicalParamNamesMap = {}
 737     for ParamName in sorted(ParamsInfo):
 738         ValidParamNames.append(ParamName)
 739         CanonicalParamNamesMap[ParamName.lower()] = ParamName
 740     
 741     # Validate paramater name and value pairs...
 742     for Index in range(0, len(ParamsOptionValueWords), 2):
 743         Name = ParamsOptionValueWords[Index]
 744         Value = ParamsOptionValueWords[Index + 1]
 745 
 746         CanonicalName = Name.lower()
 747         if  not CanonicalName in CanonicalParamNamesMap:
 748             PrintError("The parameter name, %s, specified using \"%s\" is not a valid name. Supported parameter names: %s" % (Name, ParamsOptionName, " ".join(ValidParamNames)))
 749 
 750         ParamName = CanonicalParamNamesMap[CanonicalName]
 751         ParamValue = Value
 752         
 753         if re.match("^(Sanitize|StrictParsing|RemoveHydrogens|Kekulize|SMILESIsomeric)$", ParamName, re.I):
 754             if not re.match("^(Yes|No|True|False)$", Value, re.I):
 755                 PrintError("The parameter value, %s, specified for parameter name, %s, using \"%s\" option is not a valid value. Supported values: Yes No True False" % (Value, Name, ParamsOptionName))
 756             ParamValue = True
 757             if re.match("^(No|False)$", Value, re.I):
 758                 ParamValue = False
 759         elif re.match("^SMILESTitleLine$", ParamName, re.I):
 760             if re.match("^Infile$", Mode, re.I):
 761                 if not re.match("^(Yes|No|True|False|Auto)$", Value, re.I):
 762                     PrintError("The parameter value, %s, specified for paramater name, %s, using \"%s\" option is not a valid value. Supported values: Yes No True False Auto" % (Value, Name, ParamsOptionName))
 763             elif re.match("^Outfile$", Mode, re.I):
 764                 if not re.match("^(Yes|No|True|False)$", Value, re.I):
 765                     PrintError("The parameter value, %s, specified for parameter name, %s, using \"%s\" option is not a valid value. Supported values: Yes No True False" % (Value, Name, ParamsOptionName))
 766                 ParamValue = True
 767                 if re.match("^(No|False)$", Value, re.I):
 768                     ParamValue = False
 769         elif re.match("^SMILESDelimiter$", ParamName, re.I):
 770             if not re.match("^(space|tab|comma)$", Value, re.I):
 771                 PrintError("The parameter value, %s, specified for parameter name, %s, using \"%s\" option is not a valid value. Supported values: space tab comma" % (Value, Name, ParamsOptionName))
 772             ParamValue = " "
 773             if re.match("^tab$", Value, re.I):
 774                 ParamValue = "\t"
 775             elif re.match("^comma$", Value, re.I):
 776                 ParamValue = ","
 777         elif re.match("^Compute2DCoords$", ParamName, re.I):
 778             # No need to set the value. It would be processed later to handle "auto" value...
 779             if not re.match("^(Yes|No|True|False|Auto)$", Value, re.I):
 780                 PrintError("The parameter value, %s, specified for paramater name, %s, using \"%s\" option is not a valid value. Supported values: Yes No True False Auto" % (Value, Name, ParamsOptionName))
 781         else:
 782             ParamValue = int(Value)
 783             if ParamValue <= 0:
 784                 PrintError("The parameter value, %s, specified for parameter name, %s, using \"%s\" option is not a valid value. Supported values: > 0" % (Value, Name, ParamsOptionName))
 785         
 786         # Set value...
 787         ParamsInfo[ParamName] = ParamValue
 788         
 789     # Handle paramaters with possible auto values...
 790     _ProcessInfileAndOutfileAutoParameters(Mode, ParamsInfo, ParamsOptionName, ParamsOptionValue, InfileName, OutfileName)

 792 def _ProcessInfileAndOutfileAutoParameters(Mode, ParamsInfo, ParamsOptionName, ParamsOptionValue, InfileName, OutfileName):
 793     """Process parameters with possible auto values.
 794     
 795     """
 796     if re.match("^Infile$", Mode, re.I):
 797         # SMILESTitleLine parameter...
 798         Value = ParamsInfo["SMILESTitleLine"]
 799         ParamValue = False
 800         if re.match("^auto$", Value, re.I):
 801             if InfileName is not None:
 802                 if CheckFileExt(InfileName, "smi csv tsv txt"):
 803                     ParamValue = DoesSMILESFileContainTitleLine(InfileName)
 804         elif re.match("^(Yes|True)$", Value, re.I):
 805             ParamValue = True
 806         ParamsInfo["SMILESTitleLine"] = ParamValue
 807     elif re.match("^Outfile$", Mode, re.I):
 808         # Compute2DCoords parameter...
 809         Value = ParamsInfo["Compute2DCoords"]
 810         ParamValue = False
 811         if re.match("^auto$", Value, re.I):
 812             if InfileName is not None:
 813                 if CheckFileExt(InfileName, "smi csv tsv txt"):
 814                     ParamValue = True
 815             if OutfileName is not None:
 816                 if CheckFileExt(OutfileName, "smi csv tsv txt"):
 817                     # No need to compute 2D coords for SMILES file...
 818                     ParamValue = False
 819         elif re.match("^(Yes|True)$", Value, re.I):
 820             ParamValue = True
 821         ParamsInfo["Compute2DCoords"] = ParamValue

 823 def ReplaceHTMLEntitiesInText(Text):
 824     """Check and replace the followng HTML entities to their respective code
 825     for display in a browser: < (less than), > (greater than), & (ampersand),
 826     " (double quote),  and ' (single quote).
 827 
 828     Arguments:
 829         Text (str): Text value.
 830 
 831     Returns:
 832         str : Modifed text value.
 833 
 834     """
 835 
 836     if re.search("""(<|>|&|"|')""", Text):
 837         return Text.replace("<", "&lt;").replace(">", "&gt;").replace("&", "&amp;").replace('"', "&quot;").replace("'","&apos;")
 838     else:
 839         return Text

 843     characters.
 844     
 845     Arguments:
 846         Text (string): Input text.
 847         Width (int): Max number of characters before truncating text.
 848         Delimiter (string): Trailing characters to append or None.
 849

 851         str : Truncated text
 852 
 853     """
 854     
 855     if len(Text) < Width:
 856         return Text
 857

 860     return TruncatedText
